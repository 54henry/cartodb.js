Examples
// no options
new carto.layer.Layer(citiesSource, citiesStyle);
// with options
new carto.layer.Layer(citiesSource, citiesStyle, {
  featureClickColumns: [ 'name', 'population' ],
  featureOverColumns: [ 'name' ]
});
